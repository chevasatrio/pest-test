AAA Patern (Arrange, Act, Assert) 

test('deskripsi test yang diuji', function(){

    //Arrange - siapkan kondisi awal
    $user = action();

    //Act - Lakukan aksi yang ingin diuji
    $response = $this->actingAs($user)->get('/view')

    //Assert - verifikasi hasilnya apakah sesuai dengan ekspektasi
    $response->assertStatus(200);

});
